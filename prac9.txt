#include "y.tab.h"

%%
"for"           { return FOR; }
"("             { return '('; }
")"             { return ')'; }
";"             { return ';'; }
"="             { return '='; }
"<="            { return LE; }
">="            { return GE; }
"=="            { return EQ; }
"++"            { return INC; }   /* return INC token */
"--"            { return DEC; }   /* return DEC token */
"<"             { return '<'; }
">"             { return '>'; }
"+"             { return '+'; }
"-"             { return '-'; }
[a-zA-Z_][a-zA-Z0-9_]*   { return ID; }
[0-9]+          { return NUM; }
[ \t\n]         ;   /* ignore whitespace */
.               { return yytext[0]; }
%%
int yywrap() { return 1; }
